script.js


// Function to validate input and generate multiplication table
function generateTable() {
  const input = document.getElementById("numberInput").value;
  const error = document.getElementById("error");
  const output = document.getElementById("tableOutput");

  // Clear previous output
  error.textContent = "";
  output.innerHTML = "";

  // Validation: check empty input
  if (!input) {
    error.textContent = "❌ Please enter a value.";
    return;
  }

  // Validation: check if it's a number
  if (isNaN(input)) {
    error.textContent = "❌ Please enter a valid number.";
    return;
  }

  // Convert input to number
  const num = parseInt(input);

  // Create a multiplication table
  let tableHTML = "<table border='1' cellpadding='5'>";
  for (let i = 1; i <= 10; i++) {
    tableHTML += `<tr><td>${num} × ${i}</td><td>= ${num * i}</td></tr>`;
  }
  tableHTML += "</table>";

  // Display table
  output.innerHTML = tableHTML;
}
