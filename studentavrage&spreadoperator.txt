Q. Create a simple "Student Marks & Average Calculator" web application using the following technologies:

✅ Requirements

Use Node.js with Express to create a server and handle routes.

On the homepage (index.html), create a form with:

First Name

Marks in 3 subjects (Maths, Science, English)

Submit button

Validate the form using JavaScript:

First Name should not be empty.

Marks must be numbers between 0–100.

On submit, store the marks in an array and use the spread operator (...) to:

Find the maximum and minimum marks.

Calculate the average.

Display the results dynamically on the same page (without reloading).

Style the form and result display with CSS.


server.js



// server.js
// Simple Express server to serve the static files from /public

const express = require('express');
const path = require('path');

const app = express();
const PORT = 3000;

// Serve static files from "public" folder
app.use(express.static(path.join(__dirname, 'public')));

// Optional: serve index explicitly (not required but explicit)
app.get('/', (req, res) => {
  res.sendFile(path.join(__dirname, 'public', 'index.html'));
});

// Start server
app.listen(PORT, () => {
  console.log(`✅ Server running at http://localhost:${PORT}`);
});

script.js




// script.js
// Attaches handlers after DOM loaded and calculates highest, lowest, average using spread operator.

document.addEventListener('DOMContentLoaded', () => {
  // Debug log so you can confirm the script is loaded (open browser console)
  console.log('script.js loaded');

  const form = document.getElementById('marksForm');
  const errorEl = document.getElementById('error');
  const resultsEl = document.getElementById('results');

  // Helper to escape HTML for safety
  function escapeHtml(str) {
    return String(str).replace(/[&<>"']/g, s => ({
      '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;'
    }[s]));
  }

  form.addEventListener('submit', (e) => {
    e.preventDefault(); // prevent page reload

    // Clear previous messages
    errorEl.textContent = '';
    resultsEl.innerHTML = '';

    // Read and trim inputs
    const firstName = document.getElementById('firstName').value.trim();
    const mathsRaw = document.getElementById('maths').value.trim();
    const scienceRaw = document.getElementById('science').value.trim();
    const englishRaw = document.getElementById('english').value.trim();

    // Basic validation
    if (!firstName) {
      errorEl.textContent = '❌ First Name cannot be empty';
      console.warn('Validation failed: first name empty');
      return;
    }

    if (mathsRaw === '' || scienceRaw === '' || englishRaw === '') {
      errorEl.textContent = '❌ Please enter all three marks';
      console.warn('Validation failed: missing mark(s)');
      return;
    }

    // Convert to numbers
    const maths = Number(mathsRaw);
    const science = Number(scienceRaw);
    const english = Number(englishRaw);

    // Validate numeric and range
    if ([maths, science, english].some(v => Number.isNaN(v))) {
      errorEl.textContent = '❌ Marks must be valid numbers';
      console.warn('Validation failed: non-numeric mark');
      return;
    }

    if ([maths, science, english].some(v => v < 0 || v > 100)) {
      errorEl.textContent = '❌ Marks must be between 0 and 100';
      console.warn('Validation failed: mark out of range');
      return;
    }

    // Build marks array
    const marks = [maths, science, english];

    // Use spread operator to get max & min
    const maxMark = Math.max(...marks);
    const minMark = Math.min(...marks);
    const avg = marks.reduce((a, b) => a + b, 0) / marks.length;

    // Display results (escape user text)
    resultsEl.innerHTML = `
      <h3>Results for ${escapeHtml(firstName)}</h3>
      <p><strong>Marks:</strong> ${marks.join(', ')}</p>
      <p><strong>Highest Mark:</strong> ${maxMark}</p>
      <p><strong>Lowest Mark:</strong> ${minMark}</p>
      <p><strong>Average:</strong> ${avg.toFixed(2)}</p>
    `;

    // Also log to console for debugging
    console.log('Computed:', { firstName, marks, maxMark, minMark, avg });
  });
});
