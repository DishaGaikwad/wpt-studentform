server.js


const express = require('express'); // Import Express
const bodyParser = require('body-parser'); // Import body-parser
const app = express();
const PORT = 3000;

app.use(express.static('public')); // Serve static files
app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: true }));

// POST route to generate username
app.post('/generate-username', (req, res) => {
    const { firstName, lastName, age } = req.body;

    // Server-side validation
    if (!firstName || !lastName || !age) {
        return res.json({ success: false, message: "All fields are required." });
    }

    if (firstName.length < 2 || lastName.length < 2) {
        return res.json({ success: false, message: "FirstName and LastName must have at least 2 characters." });
    }

    // Generate username: first 2 letters of firstName + last 2 letters of lastName + age
    const username = firstName.substring(0,2).toLowerCase() + lastName.slice(-2).toLowerCase() + age;

    res.json({ success: true, username: username, message: `Username generated: ${username}` });
});

app.listen(PORT, () => {
    console.log(`Server running at http://localhost:${PORT}`);
});

$(document).ready(function() {
    $('#userForm').submit(function(e) {
        e.preventDefault(); // Prevent page reload

        const firstName = $('#firstName').val().trim();
        const lastName = $('#lastName').val().trim();
        const age = $('#age').val().trim();

        // Client-side validation
        if (firstName.length < 2) {
            $('#response').text('First Name must have at least 2 characters').removeClass().addClass('error');
            return;
        }

        if (lastName.length < 2) {
            $('#response').text('Last Name must have at least 2 characters').removeClass().addClass('error');
            return;
        }

        if (!age || age <= 0) {
            $('#response').text('Enter a valid age').removeClass().addClass('error');
            return;
        }

        // Send data to server using fetch API
        fetch('/generate-username', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ firstName, lastName, age })
        })
        .then(response => response.json())
        .then(data => {
            if (data.success) {
                $('#response').text(data.message).removeClass().addClass('success');
                $('#userForm')[0].reset();
            } else {
                $('#response').text(data.message).removeClass().addClass('error');
            }
        })
        .catch(error => {
            $('#response').text('Server error').removeClass().addClass('error');
            console.error(error);
        });
    });
});
